## Step 3: Build Your First Agent

### Python Example

Create `agent.py`:

```python
import os
import requests
from dotenv import load_dotenv

# Load environment variables
load_dotenv()

class GigabrainAgent:
    def __init__(self):
        self.api_key = os.getenv("GIGABRAIN_API_KEY")
        self.base_url = "https://api.gigabrain.gg"
        
        if not self.api_key:
            raise ValueError("GIGABRAIN_API_KEY not found in environment")
    
    def analyze(self, message):
        """Send a message to Gigabrain and get analysis"""
        try:
            response = requests.post(
                f"{self.base_url}/v1/chat",
                headers={
                    "Authorization": f"Bearer {self.api_key}",
                    "Content-Type": "application/json"
                },
                json={"message": message},
                timeout=30
            )
            
            response.raise_for_status()
            return response.json()
            
        except requests.exceptions.RequestException as e:
            print(f"Error communicating with Gigabrain: {e}")
            return None
    
    def run(self):
        """Main agent loop"""
        print("Gigabrain Agent Started")
        print("-" * 50)
        
        # Example: Analyze BTC
        result = self.analyze("What is the current BTC price and trend?")
        
        if result:
            print(f"Session ID: {result.get('session_id')}")
            print(f"Analysis: {result.get('message')}")
            print(f"Timestamp: {result.get('timestamp')}")
        else:
            print("Failed to get analysis")

if __name__ == "__main__":
    agent = GigabrainAgent()
    agent.run()
```

### JavaScript Example

Create `agent.js`:

```javascript
import fetch from "node-fetch";
import dotenv from "dotenv";

// Load environment variables
dotenv.config();

class GigabrainAgent {
  constructor() {
    this.apiKey = process.env.GIGABRAIN_API_KEY;
    this.baseUrl = "https://api.gigabrain.gg";

    if (!this.apiKey) {
      throw new Error("GIGABRAIN_API_KEY not found in environment");
    }
  }

  async analyze(message) {
    try {
      const response = await fetch(`${this.baseUrl}/v1/chat`, {
        method: "POST",
        headers: {
          Authorization: `Bearer ${this.apiKey}`,
          "Content-Type": "application/json",
        },
        body: JSON.stringify({ message }),
      });

      if (!response.ok) {
        throw new Error(`HTTP error! status: ${response.status}`);
      }

      return await response.json();
    } catch (error) {
      console.error("Error communicating with Gigabrain:", error);
      return null;
    }
  }

  async run() {
    console.log("Gigabrain Agent Started");
    console.log("-".repeat(50));

    // Example: Analyze BTC
    const result = await this.analyze(
      "What is the current BTC price and trend?"
    );

    if (result) {
      console.log(`Session ID: ${result.session_id}`);
      console.log(`Analysis: ${result.message}`);
      console.log(`Timestamp: ${result.timestamp}`);
    } else {
      console.log("Failed to get analysis");
    }
  }
}

// Run the agent
const agent = new GigabrainAgent();
agent.run();
```

### Troubleshooting

**401 Unauthorized Error:**

- Check that your API key is correct
- Ensure the `.env` file is in the project root
- Verify the environment variable is loading correctly

**429 Rate Limit Error:**

- You've exceeded the rate limit (30 requests/minute)
- Wait for the reset time specified in the error
- Implement rate limiting in your agent

**Connection Errors:**

- Check your internet connection
- Verify the API endpoint URL is correct
- Check if there are any firewall restrictions

## Step 5: Add Advanced Features

### Implement Rate Limiting

**Python:**

```python
import time
from datetime import datetime, timedelta

class RateLimiter:
    def __init__(self, max_requests=30, time_window=60):
        self.max_requests = max_requests
        self.time_window = time_window
        self.requests = []
    
    def wait_if_needed(self):
        now = datetime.now()
        cutoff = now - timedelta(seconds=self.time_window)
        
        # Remove old requests
        self.requests = [req for req in self.requests if req > cutoff]
        
        if len(self.requests) >= self.max_requests:
            sleep_time = (self.requests[0] - cutoff).total_seconds()
            print(f"Rate limit reached. Waiting {sleep_time:.1f}s...")
            time.sleep(sleep_time)
        
        self.requests.append(now)
```

### Add Error Handling and Retries

**Python:**

```python
import time

def retry_with_backoff(func, max_retries=3, backoff_factor=2):
    """Retry a function with exponential backoff"""
    for attempt in range(max_retries):
        try:
            return func()
        except Exception as e:
            if attempt == max_retries - 1:
                raise
            
            wait_time = backoff_factor ** attempt
            print(f"Attempt {attempt + 1} failed: {e}")
            print(f"Retrying in {wait_time}s...")
            time.sleep(wait_time)
```

class MomentumAgent(GigabrainAgent):
    def __init__(self, symbols=["BTC", "ETH"]):
        super().__init__()
        self.symbols = symbols
        self.positions = {}
    
    def analyze_symbol(self, symbol):
        """Get analysis for a specific symbol"""
        message = f"Analyze {symbol} momentum and provide entry/exit recommendation"
        return self.analyze(message)
    
    def should_enter(self, analysis):
        """Determine if we should enter a position"""
        # Parse analysis and apply your logic
        # This is a simplified example
        if "bullish" in analysis.lower() and "high confidence" in analysis.lower():
            return True
        return False
    
    def should_exit(self, symbol, analysis):
        """Determine if we should exit a position"""
        if symbol in self.positions:
            if "bearish" in analysis.lower() or "take profit" in analysis.lower():
                return True
        return False
    
    def run_strategy(self):
        """Run the momentum strategy"""
        for symbol in self.symbols:
            result = self.analyze_symbol(symbol)
            
            if not result:
                continue
            
            analysis = result.get('message', '')
            
            if self.should_enter(analysis) and symbol not in self.positions:
                print(f"ðŸŸ¢ SIGNAL: Enter {symbol} position")

            elif self.should_exit(symbol, analysis):

## Step 7: Deploy Your Agent

### Run Continuously

**Using a loop:**

```python
import time

def run_agent_loop(agent, interval=3600):
    """Run agent in a loop with specified interval (seconds)"""
    while True:
        try:
            agent.run_strategy()
            print(f"Sleeping for {interval}s...")
            time.sleep(interval)
        except KeyboardInterrupt:
            print("Agent stopped by user")
            break
        except Exception as e:
            print(f"Error in agent loop: {e}")
            time.sleep(60)  # Wait before retrying
```
